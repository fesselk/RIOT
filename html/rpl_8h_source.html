    262 static inline uint8_t GNRC_RPL_COUNTER_INCREMENT(uint8_t counter)
    263 {
    264  return ((counter > GNRC_RPL_COUNTER_LOWER_REGION) ?
    265  ((counter == GNRC_RPL_COUNTER_MAX) ? counter = 0 : ++counter) :
    266  ((counter == GNRC_RPL_COUNTER_LOWER_REGION) ? counter = 0 : ++counter));
    267 }
    268 
    269 static inline bool GNRC_RPL_COUNTER_IS_INIT(uint8_t counter)
    270 {
    271  return (counter > GNRC_RPL_COUNTER_LOWER_REGION);
    272 }

    274 static inline bool GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(uint8_t A, uint8_t B)
    275 {
    276  return (((A < B) && (GNRC_RPL_COUNTER_LOWER_REGION + 1 - B + A < GNRC_RPL_COUNTER_SEQ_WINDOW))
    277  || ((A > B) && (A - B < GNRC_RPL_COUNTER_SEQ_WINDOW)));
    278 }
    279 
    280 static inline bool GNRC_RPL_COUNTER_GREATER_THAN(uint8_t A, uint8_t B)
    281 {
    282  return ((A > GNRC_RPL_COUNTER_LOWER_REGION) ? ((B > GNRC_RPL_COUNTER_LOWER_REGION) ?
    283  GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(A, B) : 0) :
    284  ((B > GNRC_RPL_COUNTER_LOWER_REGION) ? 1 : GNRC_RPL_COUNTER_GREATER_THAN_LOCAL(A, B)));
    285 }
