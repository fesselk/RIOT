    121 static inline uint64_t _xtimer_now64(void)
    122 {
    123  uint32_t now, elapsed;
    124 
    125  /* time sensitive since _xtimer_current_time is updated here */
    126  unsigned state = irq_disable();
    127  now = _xtimer_lltimer_now();
    128 #if XTIMER_MASK
    129  elapsed = _xtimer_lltimer_mask(now - _xtimer_lltimer_mask((uint32_t)_xtimer_current_time));
    130  _xtimer_current_time += (uint64_t)elapsed;
    131 #else
    132  elapsed = now - ((uint32_t)_xtimer_current_time & 0xFFFFFFFF);
    133  _xtimer_current_time += (uint64_t)elapsed;
    134 #endif
    135  irq_restore(state);
    136 
    137  return _xtimer_current_time;
    138 }

    164 static inline void _xtimer_spin(uint32_t offset) {
    165  uint32_t start = _xtimer_lltimer_now();
    166 #if XTIMER_MASK
    167  offset = _xtimer_lltimer_mask(offset);
    168  while (_xtimer_lltimer_mask(_xtimer_lltimer_now() - start) < offset);
    169 #else
    170  while ((_xtimer_lltimer_now() - start) < offset);
    171 #endif
    172 }
