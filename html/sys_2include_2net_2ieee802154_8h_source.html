    320 static inline eui64_t *ieee802154_get_iid(eui64_t *eui64, const uint8_t *addr,
    321  size_t addr_len)
    322 {
    323  int i = 0;
    324 
    325  eui64->uint8[0] = eui64->uint8[1] = 0;
    326 
    327  switch (addr_len) {
    328  case 8:
    329  eui64->uint8[0] = addr[i++] ^ 0x02;
    330  eui64->uint8[1] = addr[i++];
    331  eui64->uint8[2] = addr[i++];
    332  eui64->uint8[3] = addr[i++];
    333  eui64->uint8[4] = addr[i++];
    334  eui64->uint8[5] = addr[i++];
    335  eui64->uint8[6] = addr[i++];
    336  eui64->uint8[7] = addr[i++];
    337  break;
    338 
    339  case 4:
    340  eui64->uint8[0] = addr[i++] ^ 0x02;
    341  eui64->uint8[1] = addr[i++];
    342 
    343  /* Falls through. */
    344  case 2:
    345  eui64->uint8[2] = 0;
    346  eui64->uint8[3] = 0xff;
    347  eui64->uint8[4] = 0xfe;
    348  eui64->uint8[5] = 0;
    349  eui64->uint8[6] = addr[i++];
    350  eui64->uint8[7] = addr[i++];
    351  break;
    352 
    353  default:
    354  return NULL;
    355  }
    356 
    357  return eui64;
    358 }
