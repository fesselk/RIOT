    543 static inline void sercom_clk_en(void *sercom)
    544 {
    545  const uint8_t id = sercom_id(sercom);
    546 #if defined(CPU_FAM_SAMD21)
    547  PM->APBCMASK.reg |= (PM_APBCMASK_SERCOM0 << id);
    548 #elif defined (CPU_FAM_SAMD5X)
    549  if (id < 2) {
    550  MCLK->APBAMASK.reg |= (1 << (id + 12));
    551  } else if (id < 4) {
    552  MCLK->APBBMASK.reg |= (1 << (id + 7));
    553  } else {
    554  MCLK->APBDMASK.reg |= (1 << (id - 4));
    555  }
    556 #else
    557  if (id < 5) {
    558  MCLK->APBCMASK.reg |= (MCLK_APBCMASK_SERCOM0 << id);
    559  }
    560 #if defined(CPU_FAM_SAML21)
    561  else {
    562  MCLK->APBDMASK.reg |= (MCLK_APBDMASK_SERCOM5);
    563  }
    564 #endif /* CPU_FAM_SAML21 */
    565 #endif
    566 }

    615 static inline void sercom_set_gen(void *sercom, uint8_t gclk)
    616 {
    617  const uint8_t id = sercom_id(sercom);
    618  sam0_gclk_enable(gclk);
    619 #if defined(CPU_FAM_SAMD21)
    620  GCLK->CLKCTRL.reg = (GCLK_CLKCTRL_CLKEN | GCLK_CLKCTRL_GEN(gclk) |
    621  (SERCOM0_GCLK_ID_CORE + id));
    622  while (GCLK->STATUS.reg & GCLK_STATUS_SYNCBUSY) {}
    623 #elif defined(CPU_FAM_SAMD5X)
    624  GCLK->PCHCTRL[_sercom_gclk_id_core(id)].reg = (GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN(gclk));
    625 #else
    626  if (id < 5) {
    627  GCLK->PCHCTRL[SERCOM0_GCLK_ID_CORE + id].reg = (GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN(gclk));
    628  }
    629 #if defined(CPU_FAM_SAML21)
    630  else {
    631  GCLK->PCHCTRL[SERCOM5_GCLK_ID_CORE].reg = (GCLK_PCHCTRL_CHEN | GCLK_PCHCTRL_GEN(gclk));
    632  }
    633 #endif /* CPU_FAM_SAML21 */
    634 #endif
    635 }
