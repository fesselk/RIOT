    171  if (timeout_duration <= timeout_duration.zero()) {
    172  return cv_status::timeout;
    173  }
    174  timex_t timeout, before, after;
    175  auto s = duration_cast<seconds>(timeout_duration);
    176  timeout.seconds = s.count();
